npm install --recursive npm update --recursive

Today I watched the Holowaychuk video on vimeo about modular apps. http://vimeo.com/56166857 He showed how he thought in the project main package.json that bundledDependencies should be able to have their deps recursively installed with npm install. It's still that way, which it is as of 4/16/2013, on node 0.10.4 and npm 1.2.18, win7-64

Looking in the npm tracker at similar issues, there was the non-recursive nature of update. https://github.com/isaacs/npm/issues/1341 Before getting into the npm api, if ever, I wanted to do a sanity check that recursive install would not cause a probleme extraordinaire.

The script that runs npm here can be used for scripting recursive execution of other programs as well

Usage: start_dir = "/dev/scratch" require("./exec_util").npm_install_rec(start_dir)

File exec_util.coffee:

###
Execute arbitrary args in a dir, singly or recursively.
A list of ignore files prevents bad recursions.
A list of target files triggers exec of args if found in the cwd.
Usage: See npm_install_rec below.
DEF_IGS works for me, but do you ever need to go to node_modules?
Note: file is passed but not currently used in exec_args_in_dir.
Paul McCulloch, 16 April 2013
License: <= MIT
###

fs = require "fs"
DEF_IGS = ["node_modules", ".git"]

exec_args_in_dir = (args, dir, file) ->
  console.log "exec", args, "in", dir
  exec = require('child_process').exec
  command = exec args, {cwd:dir, stio:"inherit"}, (err,stout,sterr) ->
    console.log "\n cwd?", dir,
                "\n err?", err,
                "\n stout?", stout,
                "\n sterr?", sterr

func_walk = (args, file, parent=null, ignores=DEF_IGS, targets=[]) ->
  if args? and file?
    if file in targets and parent?
      exec_args_in_dir args, parent, file
    else if file not in ignores
      if parent?
        file = parent + "/" + file
      fs.stat file, (err,stats) ->
        console.log err if err?
        if stats.isDirectory()
          fs.readdir file, (err,files) ->
            if err?
              console.log err
            else
              for item in files
                do -> func_walk(args, item, file, ignores, targets)

npm_install_rec = (start) ->
  func_walk "npm -s install", start, null, DEF_IGS, ["package.json"]

npm_update_rec = (start) ->
  func_walk "npm -s update", start, null, DEF_IGS, ["package.json"]

module.exports = exports =
  func_walk: func_walk
  exec_args_in_dir: exec_args_in_dir
  npm_install_rec: npm_install_rec
  npm_update_rec: npm_update_rec

Could be useful for an installer if npm is not the way to go and shrinkwrap doesn't do it. The problem for the bundledDependencies scenario is that it's not just "npm install" anymore, but you could use this from a Cakefile etc.
